    Model = "rbxassetid://12913890947",
    Speed = 50,
    MoveDelay = 5,
    HeightOffset = 1,
    CanKill = false,
    KillRange = 100,
    SpawnInFront = false,
    ShatterLights = false,
    FlickerLights = {
        Enabled = true,
        Duration = 100
    },
    Cycles = {
        Min = 2,
        Max = 5,
        Delay = 2
    },
    CamShake = {

entity.Debug.OnLookAtEntity = function()

Spawner.runEntity(entity)
